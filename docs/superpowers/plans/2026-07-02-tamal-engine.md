# Tamal Engine (`Engine.step`) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `Tamal.Engine` — the pure Mealy transition `step :: State -> BusIn -> (State, BusOut, Maybe Ring)` that composes every existing pure leaf into a running eSPI shift engine (fetch/decode/execute, datapath, SCK-timed bus FSM, trace ring, HALT/TRAP lifecycle).

**Architecture:** A single pure `step` dispatches on a `phase :: Phase` field to per-phase helpers (`stepIdle`, `stepPreamble`, `stepFetch`, `stepExec`, `stepBusBeat`, `stepTraceEmit`, `stepWaitAlert`, `stepHalted`). All pin drives are held in `State` and projected to `BusOut`. SCK is `fabric/5`, generated by a `busPhase :: Index 5` counter inside `step`, so essentially all bus timing is property-testable. Tests use a pure-fold cosimulation driver + an instantaneous reference interpreter (`refRun`).

**Tech Stack:** Clash 1.10 (`clash-prelude`), tasty + tasty-hedgehog + tasty-hunit, `clash-prelude-hedgehog`. Build/test with `cabal` from `hdl/`.

```haskell
-- SPDX-FileCopyrightText: 2026 Felipe Balbi
-- SPDX-License-Identifier: CERN-OHL-P-2.0

{- |
The tamal engine: a pure Mealy transition that composes every leaf into a
running eSPI shift engine (design doc 2026-07-02-tamal-engine-design.md).
'step' dispatches on 'phase' to per-phase helpers; all pin drives live in
'State' and are projected by 'busOut'.
-}
module Tamal.Engine
  ( Phase (..)
  , Pending (..)
  , State (..)
  , BusIn (..)
  , BusOut (..)
  , Ring (..)
  , initState
  , powerUpDefault
  , revisionWord
  , busOut
  , step
  ) where

import Clash.Prelude
import Tamal.Bus.Serdes (Lanes, hiZ)
import Tamal.Config (AlertSource (..), Config (..), IoMode (..), Role (..), Sck (..))
import Tamal.Isa (Reg)
import Tamal.RegFile (Regs, initRegs)

-- | Program-address width (word index): 1024-word store, matches the ±1024
-- instruction branch range (§ D3).
type AW = 10

data Phase
  = Idle | Preamble | Fetch | Exec
  | BusBeat | TraceEmit | WaitAlert | Halted
  deriving stock (Generic, Show, Eq)
  deriving anyclass (NFDataX)

-- | Work deferred to a completion phase (§3.4).
data Pending
  = PendNone
  | PendGet Reg (Unsigned 4) Bool -- rd, nbits, update RX CRC?
  | PendMark (BitVector 32)       -- MARK payload emitted by TraceEmit
  deriving stock (Generic, Show, Eq)
  deriving anyclass (NFDataX)

data State = State
  { phase :: Phase
  , pc :: Unsigned AW
  , regs :: Regs
  , cfg :: Config
  , rxCrc :: BitVector 8
  , ringPtr :: Unsigned 12
  , ovf :: Bool
  , csN :: Bit
  , sck :: Bit
  , rstN :: Bit
  , lanes :: Lanes
  , busPhase :: Index 5
  , beatIx :: Unsigned 4
  , beatTot :: Unsigned 4
  , shifter :: BitVector 8
  , pending :: Pending
  , waitTimer :: BitVector 9
  }
  deriving stock (Generic, Show, Eq)
  deriving anyclass (NFDataX)

data BusIn = BusIn
  { instrWord :: BitVector 32
  , ioIn :: Vec 4 Bit
  , alertIn :: Bit
  , startIn :: Bool
  }
  deriving stock (Generic, Show, Eq)
  deriving anyclass (NFDataX)

data BusOut = BusOut
  { pcOut :: Unsigned AW
  , csOut :: Bit
  , sckOut :: Bit
  , rstOut :: Bit
  , lanesOut :: Lanes
  , haltedOut :: Bool
  }
  deriving stock (Generic, Show, Eq)
  deriving anyclass (NFDataX)

data Ring = Ring
  { rAddr :: Unsigned 12
  , rData :: BitVector 32
  }
  deriving stock (Generic, Show, Eq)
  deriving anyclass (NFDataX)

-- | Power-up config default (§7.2): controller / x1 / 20 MHz / ALERT# pin.
powerUpDefault :: Config
powerUpDefault = Config Controller X1 Sck20 AlertPin

-- | REVISION word [major8 | minor8 | patch16] = v0.1.0 (tracks tamal.cabal).
revisionWord :: BitVector 32
revisionWord = 0x0001_0000

-- | Power-up state: Idle, pins safe, config default, ring pointer at 1.
initState :: State
initState =
  State
    { phase = Idle
    , pc = 0
    , regs = initRegs
    , cfg = powerUpDefault
    , rxCrc = 0
    , ringPtr = 1
    , ovf = False
    , csN = 1 -- deasserted
    , sck = 0 -- idle low
    , rstN = 1 -- deasserted
    , lanes = hiZ
    , busPhase = 0
    , beatIx = 0
    , beatTot = 0
    , shifter = 0
    , pending = PendNone
    , waitTimer = 0
    }

-- | Project the registered pin state to the output record.
busOut :: State -> BusOut
busOut s =
  BusOut
    { pcOut = pc s
    , csOut = csN s
    , sckOut = sck s
    , rstOut = rstN s
    , lanesOut = lanes s
    , haltedOut = phase s == Halted
    }

-- | Soft-init on 'startIn' (§8 / D9): reset run state, keep no reset port.
softInit :: State
softInit = initState {phase = Preamble}

step :: State -> BusIn -> (State, BusOut, Maybe Ring)
step s inp = case phase s of
  Idle -> stepIdle s inp
  Halted -> stepHalted s inp
  _ -> (s, busOut s, Nothing) -- filled in by later tasks

-- | Idle: hold safe pins; on 'startIn', soft-init into Preamble.
stepIdle :: State -> BusIn -> (State, BusOut, Maybe Ring)
stepIdle s inp
  | startIn inp = (softInit, busOut s, Nothing)
  | otherwise = (s, busOut s, Nothing)

-- | Halted: hold; a fresh 'startIn' re-runs via soft-init.
stepHalted :: State -> BusIn -> (State, BusOut, Maybe Ring)
stepHalted s inp
  | startIn inp = (softInit, busOut s, Nothing)
  | otherwise = (s, busOut s, Nothing)
```

```haskell
-- A run result: final state + ring writes (in emission order) + cycles used.
data Run = Run
  { runState :: State
  , runRing :: [Ring]
  , runCycles :: Int
  }

-- Fold `step` for up to `budget` cycles or until Halted. `mem` is the program;
-- `slave` supplies ioIn per cycle; alert/start are scripted by `inputs`.
drive ::
  Int ->                          -- cycle budget (watchdog)
  (Unsigned 10 -> BitVector 32) -> -- program memory
  (Int -> (Vec 4 Bit, Bit, Bool)) -> -- per-cycle (ioIn, alert, start)
  Run
drive budget mem inputs = go 0 initState 0 []
  where
    go !t !s !prevPc !ring
      | phase s == Halted && t > 0 = Run s (reverse ring) t
      | t >= budget = Run s (reverse ring) t
      | otherwise =
          let (io, alert, start) = inputs t
              inp = BusIn (mem prevPc) io alert start
              (s', bo, mr) = step s inp
              ring' = maybe ring (: ring) mr
           in go (t + 1) s' (pcOut bo) ring'

-- Convenience: run a fixed program, all-zero io, start pulsed at cycle 0.
runProg :: Int -> [BitVector 32] -> Run
runProg budget prog = drive budget mem inputs
  where
    v = fromList (prog <> repeat 0) :: Vec 1024 (BitVector 32)
    mem a = v !! a
    inputs 0 = (repeat 0, 0, True)
    inputs _ = (repeat 0, 0, False)
```

```haskell
import Tamal.Isa (Instr (..), decode, DecodeError)
import Tamal.Alu (dataResult)
import Tamal.RegFile (readReg, writeReg)
import Tamal.Trace (Record (..), encodeRecord)

-- Advance to the next sequential instruction.
advance :: State -> State
advance s = s {phase = Fetch, pc = pc s + 1}

-- Fixed terminator address for this ring depth (D words → termAddr = D-1).
termAddr :: Unsigned 12
termAddr = maxBound -- top of the ring-BRAM address space; top-shell sizes it

-- Emit the HALT terminator (§7.4). `trap`/`reason` distinguish TRAP from HALT.
haltWith :: Bool -> BitVector 3 -> BitVector 8 -> State -> (State, BusOut, Maybe Ring)
haltWith trap reason status s =
  let s' = s {phase = Halted}
      w = head (encodeRecord (Halt trap reason (ovf s) status))
   in (s', busOut s', Just (Ring termAddr w))

stepExec :: State -> BusIn -> (State, BusOut, Maybe Ring)
stepExec s inp = case decode (instrWord inp) of
  Left _ -> haltWith True 1 0 (safePins s) -- decode error → reason 1
  Right i -> execInstr i s inp

-- Drive pins safe (used by TRAP): CS# high, hiZ, SCK low, RESET# high.
safePins :: State -> State
safePins s = s {csN = 1, sck = 0, rstN = 1, lanes = hiZ}

execInstr :: Instr -> State -> BusIn -> (State, BusOut, Maybe Ring)
execInstr i s inp = case i of
  -- DATA-compute: writeback dataResult, advance.
  LoadImm rd _ -> dataWb rd
  Lui rd _ -> dataWb rd
  Mov rd _ -> dataWb rd
  Add rd _ _ -> dataWb rd
  Addi rd _ _ -> dataWb rd
  Sub rd _ _ -> dataWb rd
  And_ rd _ _ -> dataWb rd
  Andi rd _ _ -> dataWb rd
  Or_ rd _ _ -> dataWb rd
  Ori rd _ _ -> dataWb rd
  Xor_ rd _ _ -> dataWb rd
  Xori rd _ _ -> dataWb rd
  Shift rd _ _ _ -> dataWb rd
  -- CTRL: HALT (status in imm[7:0]).
  Halt st -> haltWith False 0 st s
  _ -> (advance s, busOut s, Nothing) -- other opcodes: later tasks
  where
    rs1v = readReg (regs s) (operandRs1 i)
    rs2v = readReg (regs s) (operandRs2 i)
    dataWb rd =
      let s' = (advance s) {regs = writeReg (regs s) rd (dataResult i rs1v rs2v)}
       in (s', busOut s', Nothing)
```

-- | One fabric cycle of a bus beat: drive SCK from busPhase, sequence beats.
stepBusBeat :: State -> BusIn -> (State, BusOut, Maybe Ring)
stepBusBeat s inp
  | busPhase s < 4 = (tick s, busOut (tick s), Nothing)     -- mid-beat: just advance phase
  | beatIx s + 1 < beatTot s = (nextBeat s, busOut (nextBeat s), Nothing) -- more beats
  | otherwise = complete s inp                              -- last beat done
  where
    -- advance the SCK phase, refresh SCK level, keep lane drive for the beat
    tick t = t {busPhase = busPhase t + 1, sck = sckOf (busPhase t + 1)}
    -- wrap to the next beat: phase 0, beatIx+1, refresh driven lane (PUT)
    nextBeat t =
      let bi = beatIx t + 1
       in t {busPhase = 0, beatIx = bi, sck = 0, lanes = putLanes t bi}

-- SCK level for a phase: low {0,1,2}, high {3,4}.
sckOf :: Index 5 -> Bit
sckOf p = if p >= 3 then 1 else 0

-- Lane drive for PUT beat `bi`; GET/TAR override this in later tasks.
putLanes :: State -> Unsigned 4 -> Lanes
putLanes t bi = serializeX1 (shifter t) !! bi

- **Thread `inp` into the bus helpers.** `stepBusBeat s inp` has `inp` in scope; define `tick`/`nextBeat`/`sampleGet`/`beatLanes`/`complete` so they can read `ioIn inp` for GET sampling.
